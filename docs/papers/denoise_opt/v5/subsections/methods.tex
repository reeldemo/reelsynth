\section{Methods}
\label{sec:methods}

\subsection{Notation}
\label{sec:notation}

Throughout, $x\in\mathbb{R}^{L}$ is the stored single-period wavetable (raw engine input).
$y=\Theta(x;\theta)$ is the engine-baked cycle after seam ops and optional residual cells with parameters $\theta$.
$r^{\star}\in\mathbb{R}^{L}$ is a procedural ideal sibling from the same seed and family with the open-wrap cliff withheld (not a studio-clean recording).
Tiling length $N$ (typically $N{=}16$) forms $y_{\mathrm{tiled}}=\mathrm{Tile}(y,N)$ and $r^{\star}_{\mathrm{tiled}}=\mathrm{Tile}(r^{\star},N)$.
Seam width $W$ (code: \texttt{SEAM\_W}, typically $8$) is the neighborhood edited by classical fade and polish ops.
$R\in[0,1]$ is the prolonged residual score defined below ($1$~=~best).
These symbols are fixed for the residual objective. They do not follow the Noise2Noise clean/noisy pair convention.

\subsection{Ideal tile construction}
\label{sec:ideal-tile}

The ideal sibling is generated by the same procedural draw as the engine input (frequency, phase, mid-cycle noise policy) with the open-wrap cliff amplitude set to zero at both ends.
Engine and ideal therefore share the mid-cycle content of one seed draw.
They differ by the seam cliff (and any seam-boosted noise policy applied only to the engine path).
This sibling relationship is what makes unsupervised ranking possible without paired studio recordings~\cite{lehtinen2018,kashyap2021n2n}.

\subsection{Residual score $R$}
\label{sec:residual-score}

With tiling length $N$,
\begin{equation}
\label{eq:R}
R=\mathrm{clamp}\!\left(1-\frac{\mathrm{rms}(y_{\mathrm{tiled}}-r^{\star}_{\mathrm{tiled}})}{\max(\mathrm{rms}(r^{\star}_{\mathrm{tiled}}),\varepsilon)},\,0,\,1\right),
\end{equation}
where $\mathrm{rms}(z)=\sqrt{\frac{1}{|z|}\sum_i z_i^2}$ and $\varepsilon>0$ avoids division by zero.
Optional soft shape gates (when enabled) multiply $R$ by factors that down-weight mid-cycle damage relative to seam-local error. The propositions below concern the ungated score in~\eqref{eq:R}.

\begin{proposition}[Range]
\label{prop:r-range}
For any finite $y,r^{\star}$ and $\varepsilon>0$, the score in~\eqref{eq:R} satisfies $R\in[0,1]$.
\end{proposition}
\begin{proof}
The ratio of nonnegative RMS terms is nonnegative, so $1-\mathrm{rms}(\cdot)/\max(\cdot,\varepsilon)\le 1$.
The outer $\mathrm{clamp}(\cdot,0,1)$ forces the lower bound $0$.
\end{proof}

\begin{proposition}[Perfect match]
\label{prop:perfect}
If $y_{\mathrm{tiled}}=r^{\star}_{\mathrm{tiled}}$, then the pre-clamp identity equals $1$, hence $R=1$.
\end{proposition}
\begin{proof}
$\mathrm{rms}(0)=0$, so $1-0/\max(\mathrm{rms}(r^{\star}_{\mathrm{tiled}}),\varepsilon)=1$, and clamp leaves $1$.
\end{proof}

\begin{proposition}[Monotone RMS]
\label{prop:monotone}
Fix $r^{\star}_{\mathrm{tiled}}$. If $\mathrm{rms}(y_{\mathrm{tiled}}-r^{\star}_{\mathrm{tiled}})<\mathrm{rms}(y'_{\mathrm{tiled}}-r^{\star}_{\mathrm{tiled}})$ and both pre-clamp scores lie in $(0,1)$, then the pre-clamp score of $y$ is strictly larger than that of $y'$.
\end{proposition}
\begin{proof}
The map $e\mapsto 1-e/d$ with fixed $d=\max(\mathrm{rms}(r^{\star}_{\mathrm{tiled}}),\varepsilon)>0$ is strictly decreasing in the error RMS $e$.
\end{proof}

\begin{lemma}[Wrap closure, sufficient sketch]
\label{lem:wrap}
Suppose the only difference between engine and ideal is an endpoint cliff, and a bake zeros the endpoint jump while changing mid-cycle samples by at most $\varepsilon_{\mathrm{mid}}$ in RMS.
Then $R$ is nondecreasing relative to the untreated engine whenever the mid-cycle penalty is dominated by the seam error reduction under~\eqref{eq:R}.
\end{lemma}
\begin{proof}[Sketch]
This is a conditional accounting argument under the stated assumptions, not a claim that every DualCosine-style fade improves $R$ on every draw.
Empirical support appears in the classical and favorite benches of Section~\ref{sec:results}.
\end{proof}

\begin{remark}[Noise2Noise motivation]
\label{rem:n2n}
Ranking without studio-clean pairs is possible when the ideal is a procedural sibling of the engine draw~\cite{lehtinen2018,kashyap2021n2n}.
This is a design philosophy, not a theorem about speech corpora.
\end{remark}

\begin{algorithm}[t]
\caption{ResidualScore}
\label{alg:residual}
\begin{algorithmic}[1]
\Require Ideal cycle $r^{\star}$, baked cycle $y$, periods $N$
\State $I \gets \mathrm{Tile}(r^{\star}, N)$
\State $Y \gets \mathrm{Tile}(y, N)$
\State $r_{\mathrm{rms}} \gets \mathrm{RMS}(Y-I)$
\State $i_{\mathrm{rms}} \gets \max(\mathrm{RMS}(I), \varepsilon)$
\State \Return $\mathrm{clamp}(1 - r_{\mathrm{rms}}/i_{\mathrm{rms}},\,0,\,1)$
\end{algorithmic}
\end{algorithm}

\subsection{Bake operator and DualCosine baseline}
\label{sec:bake}

A \textbf{bake cell} $\Theta(\cdot;\theta)$ composes classical seam ops (DualCosine, polish, soft seam, FIR blends) with optional tiny residual cells (MLP/FIR/U-Net-lite/attention-lite) and optional MoE soft gates over experts.
Bake parameters $\theta$ live in a bounded box. Architecture strings are discrete graphs over those ops.

\begin{algorithm}[t]
\caption{DualCosine baseline}
\label{alg:dualcosine}
\begin{algorithmic}[1]
\Require Engine cycle $x$
\State Fade both ends of $x$ with raised-cosine windows of width $W$
\State \Return $\mathrm{ResidualScore}(r^{\star},\,x_{\mathrm{faded}},\,N)$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{MoE soft gating over bake experts}
\label{alg:moe}
\begin{algorithmic}[1]
\Require Cycle $x$, expert cells $\{E_k\}_{k=1}^{K}$, gate logits $g\in\mathbb{R}^{K}$
\State $w \gets \mathrm{softmax}(g)$
\State $y \gets \sum_{k=1}^{K} w_k\, E_k(x)$
\State \Return $y$
\end{algorithmic}
\end{algorithm}

\subsection{Search space}
\label{sec:search-space}

Each genome encodes: ordered op list (fade, polish, FIR, pin, residual write), depth (number of composed blocks), residual cell type (none, MLP, FIR, U-Net-lite, attn-lite), optional MoE expert count $K\in\{2,3,4\}$, and continuous $\theta$ in box bounds used by the overnight runner (fade widths, FIR taps, residual scales).
The live tag is \texttt{PPO+GA+PBT+NAS+depth+MoE}.

\subsection{Training / fit inner loop}
\label{sec:fit}

\begin{algorithm}[t]
\caption{FitCell (inner loop)}
\label{alg:fit}
\begin{algorithmic}[1]
\Require Architecture cfg, fit steps $T$, batch size $B$
\State cell $\gets \mathrm{SeamCell}(\mathrm{cfg})$; opt $\gets \mathrm{Adam}(\mathrm{cell.params})$
\State prev $\gets \mathrm{null}$; patience $\gets 0$
\For{$t = 1,\ldots,T$}
  \State ideal, eng $\gets \mathrm{MakeBatch}(B)$
  \State out $\gets \mathrm{ApplyOps}(\mathrm{eng},\,\mathrm{cell},\,\mathrm{cfg.ops})$
  \State $R \gets \mathrm{mean}\,\mathrm{ResidualScore}(\mathrm{ideal},\,\mathrm{out})$
  \State loss $\gets 1-R$
  \If{finite(loss)} AdamStep(opt, loss)
  \EndIf
  \If{prev $\neq$ null and $|\mathrm{prev}-R|/\max(|\mathrm{prev}|,\varepsilon) < 10^{-4}$}
    \State patience $\gets$ patience $+1$; \textbf{if} patience $\ge 3$ \textbf{then break}
  \Else
    \State patience $\gets 0$
  \EndIf
  \State prev $\gets R$
\EndFor
\State \Return cell, $R$
\end{algorithmic}
\end{algorithm}

\subsection{Hybrid outer loop}
\label{sec:hybrid}

Each iteration rotates among branches: PPO, GA, PBT, NAS, and combo.
Selection is always by residual $R$ (plus logged auxiliaries).
Reward for PPO is $R$ minus the DualCosine baseline on the same batch.

\begin{algorithm}[t]
\caption{GA tournament step}
\label{alg:ga}
\begin{algorithmic}[1]
\Require Population $P$, tournament size $k$, crossover rate $p_c$, mutation rate $p_m$
\State Sample $k$ genomes; parents $\gets$ top-2 by $R$
\State child $\gets$ Crossover(parents) with prob.\ $p_c$, else clone elite
\State Mutate(child) with rate $p_m$ (ops, depth, cell type, $\theta$)
\State Fit and evaluate child; insert into $P$ (replace worst if full)
\State \Return updated $P$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{PPO architecture update}
\label{alg:ppo}
\begin{algorithmic}[1]
\Require Actor--critic $\pi_\phi$, clip $\epsilon$, experience buffer of mutate actions
\State Sample action $a$ editing ops / hyperparameters from $\pi_\phi$
\State Fit proposed cfg; observe reward $\rho = R - R_{\mathrm{DualCosine}}$
\State Advantage $\hat{A}$ from critic; ratio $r_\phi = \pi_\phi(a)/\pi_{\phi_{\mathrm{old}}}(a)$
\State Update $\phi$ with clipped surrogate $\min(r_\phi\hat{A},\,\mathrm{clip}(r_\phi,1-\epsilon,1+\epsilon)\hat{A})$
\State \Return updated $\pi_\phi$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{PBT exploit--mutate}
\label{alg:pbt}
\begin{algorithmic}[1]
\Require Population $P$, exploit quantile $q$, mutate scale $\sigma$
\State Rank $P$ by $R$; for each bottom member, copy hyperparameters from a top-$q$ elite
\State Perturb copied hyperparameters by multiplicative noise of scale $\sigma$
\State Refit briefly; write back into $P$
\State \Return updated $P$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{HybridMetaSearch}
\label{alg:hybrid}
\begin{algorithmic}[1]
\Require Iteration budget $T_{\mathrm{out}}$, population size $n$
\State $P \gets \mathrm{InitPopulation}(n)$; $\pi \gets \mathrm{ActorCritic}()$; champ $\gets$ null; boredom $\gets 0$
\For{$it = 1,\ldots,T_{\mathrm{out}}$}
  \State branch $\gets \mathrm{Rotate}(\mathrm{ppo},\mathrm{ga},\mathrm{pbt},\mathrm{nas},\mathrm{combo})$
  \State cfg, hp $\gets \mathrm{Propose}(\mathrm{branch},\,P,\,\pi)$
  \State cell, $R_{\mathrm{fit}} \gets \mathrm{FitCell}(\mathrm{cfg},\,\mathrm{hp.fit\_steps})$
  \State $R_{\mathrm{eval}} \gets \mathrm{EvalCell}(\mathrm{cell})$
  \State $R \gets \tfrac12 R_{\mathrm{fit}} + \tfrac12 R_{\mathrm{eval}}$ (+ optional depth/MoE bonus)
  \State UpdatePopulation($P$, cfg, $R$); PPOUpdate($\pi$, reward $= R - R_{\mathrm{DualCosine}}$)
  \If{$R > \mathrm{champ}.R$} champ $\gets$ (cfg, cell, $R$); boredom $\gets 0$
  \Else\ boredom $\gets$ boredom $+1$
  \EndIf
  \If{boredom $\ge 1000$} PlateauAdapt; boredom $\gets 0$
  \EndIf
\EndFor
\State \Return champ
\end{algorithmic}
\end{algorithm}

\begin{remark}[Trial complexity]
\label{rem:complexity}
Per outer iteration the dominant cost is $O(T_{\mathrm{fit}}\cdot C_{\mathrm{fwd}})$ for fit steps times one forward bake.
Bake-width caps (tiny residual cells) keep $C_{\mathrm{fwd}}$ far below full-band Demucs/DiffWave forwards screened in Related Work.
\end{remark}
